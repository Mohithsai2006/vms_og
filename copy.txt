mkdir wordpress-mysql-app

version: '3.8'

services:
  db:
    image: mysql:8.0
    container_name: mysql_db
    restart: unless-stopped
    # Set environment variables for MySQL (Crucial for initial setup)
    environment:
      MYSQL_ROOT_PASSWORD: your_root_password # CHANGE THIS
      MYSQL_DATABASE: wordpress
      MYSQL_USER: wordpressuser # CHANGE THIS
      MYSQL_PASSWORD: your_db_password # CHANGE THIS
    volumes:
      - db_data:/var/lib/mysql # Persistent storage for database data
    networks:
      - wp-network

  wordpress:
    depends_on:
      - db # Ensure the database starts before WordPress
    image: wordpress:latest
    container_name: wordpress_app
    restart: unless-stopped
    # Map port 80 in the container to port 8080 on your host machine
    ports:
      - "8085:80" 
    # Set environment variables for WordPress to connect to MySQL
    environment:
      WORDPRESS_DB_HOST: db:3306 # 'db' is the service name of the MySQL container
      WORDPRESS_DB_USER: wordpressuser # Must match MYSQL_USER above
      WORDPRESS_DB_PASSWORD: your_db_password # Must match MYSQL_PASSWORD above
      WORDPRESS_DB_NAME: wordpress
    volumes:
      - wp_content:/var/www/html/wp-content # Persistent storage for themes/plugins/uploads
    networks:
      - wp-network

volumes:
  db_data: # Define a volume for the MySQL database files
  wp_content: # Define a volume for WordPress content (themes, plugins, uploads)

networks:
  wp-network: # Define a custom network for containers to communicate

7.JENKINS SCRIPTED PIPELINE JAVAPROJECT

node {

    stage('Clone Repository') {
        git branch: 'main',
            url: 'https://github.com/budarajumadhurika/SampleMavenJavaProject.git'
    }

    stage('Build') {
        bat 'mvn clean package'
    }

    stage('Test') {
        bat 'mvn test'
    }

    stage('Archive') {
        archiveArtifacts artifacts: 'target/*.jar', fingerprint: true
        junit 'target/surefire-reports/*.xml'
    }
}


8.WEBPROJECT SCRIPTED PIPELINE JENKINS

node {

    stage('Clone Repository') {
        git branch: 'main',
            url: 'https://github.com/budarajumadhurika/SampleMavenWebProject.git'
    }

    stage('Build WAR') {
        bat 'mvn clean package'
    }

    stage('Run Tests') {
        bat 'mvn test'
    }

    stage('Archive WAR & Test Results') {

        // Archive WAR file
        archiveArtifacts artifacts: 'target/*.war', fingerprint: true

        // If test reports exist → archive them, otherwise skip
        if (fileExists('target/surefire-reports')) {
            junit 'target/surefire-reports/*.xml'
        } else {
            echo "No test reports found — skipping junit archiving."
        }
    }

    stage('Summary') {
        echo "Build completed successfully!"
    }
}

node {
    properties([
        pipelineTriggers([
            [$class: 'GitHubPushTrigger']
        ])
    ])

    stage('Checkout') {
        checkout([$class: 'GitSCM',
            branches: [[name: '*/main']],
            userRemoteConfigs: [[
                url: 'https://github.com/Mohithsai2006/webhook.git',
                credentialsId: '1a55c9f2-4b4f-4ea0-8f81-7b03baf86858'
            ]]
        ])
    }

    stage('Build') {
        echo "Webhook triggered pipeline build!"
    }
}
